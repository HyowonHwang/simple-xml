019package org.simpleframework.xml.filter;
020
021/**
022 * The <code>SystemFilter</code> object is used to provide a filter
023 * that will replace the specified values with system properties.
024 * This can be given a delegate filter which can be used to resolve
025 * replacements should the value requested not match a property. 
026 * 
027 * @author Niall Gallagher
028 */
029public class SystemFilter implements Filter {
030   
031   /**
032    * Filter delegated to if no system property can be resolved.
033    */
034   private Filter filter;        
035        
036   /**
037    * Constructor for the <code>SystemFilter</code> object. This 
038    * creates a filter that will resolve replacements using system
039    * properties. Should the system properties not contain the
040    * requested mapping this will return a null value.
041    */
042   public SystemFilter() {
043      this(null);           
044   }
045        
046   /**
047    * Constructor for the <code>SystemFilter</code> object. This 
048    * creates a filter that will resolve replacements using system
049    * properties. Should the system properties not contain the
050    * requested mapping this delegates to the specified filter.
051    * 
052    * @param filter the filter delegated to if resolution fails
053    */   
054   public SystemFilter(Filter filter) {
055      this.filter = filter;           
056   }
057   
058   /**
059    * Replaces the text provided with the value resolved from the
060    * system properties. If the system properties fails this will
061    * delegate to the specified <code>Filter</code> if it is not
062    * a null object. If no match is found a null is returned.
063    * 
064    * @param text this is the text value to be replaced
065    * 
066    * @return this will return the replacement text resolved
067    */
068   public String replace(String text) {
069      String value = System.getProperty(text);           
070
071      if(value != null) {
072         return value;                       
073      }
074      if(filter != null) {
075         return filter.replace(text);              
076      }      
077      return null;
078   }   
079}
