019package org.simpleframework.xml.filter;
020
021import java.util.Stack;
022
023/**
024 * The <code>StackFilter</code> object provides a filter that can
025 * be given a collection of filters which can be used to resolve a
026 * replacement. The order of the resolution used for this filter
027 * is last in first used. This order allows the highest priority
028 * filter to be added last within the stack. 
029 * 
030 * @author Niall Gallagher 
031 */
032public class StackFilter implements Filter {
033
034   /**
035    * This is used to store the filters that are used.
036    */
037   private Stack<Filter> stack;        
038        
039   /**
040    * Constructor for the <code>StackFilter</code> object. This will
041    * create an empty filter that initially resolves null for all
042    * replacements requested. As filters are pushed into the stack
043    * the <code>replace</code> method can resolve replacements. 
044    */
045   public StackFilter() {
046      this.stack = new Stack<Filter>();
047   }
048
049   /**
050    * This pushes the the provided <code>Filter</code> on to the top
051    * of the stack. The last filter pushed on to the stack has the
052    * highes priority in the resolution of a replacement value.
053    * 
054    * @param filter this is a filter to be pushed on to the stack
055    */
056   public void push(Filter filter) {
057      stack.push(filter);           
058   }
059   
060   /**
061    * Replaces the text provided with the value resolved from the
062    * stacked filters. This attempts to resolve a replacement from
063    * the top down. So the last <code>Filter</code> pushed on to
064    * the stack will be the first filter queried for a replacement.
065    * 
066    * @param text this is the text value to be replaced
067    * 
068    * @return this will return the replacement text resolved
069    */
070   public String replace(String text) {
071      for(int i = stack.size(); --i >= 0;) {
072         String value = stack.get(i).replace(text);
073
074         if(value != null){
075            return value;                 
076         }         
077      }           
078      return null;
079   }
080}
