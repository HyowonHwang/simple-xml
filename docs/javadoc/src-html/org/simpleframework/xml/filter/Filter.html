019package org.simpleframework.xml.filter;
020
021/**
022 * The <code>Filter</code> object is used to provide replacement string
023 * values for a provided key. This allows values within the XML source
024 * document to be replaced using sources such as OS environment variables
025 * and Java system properties.
026 * <p>
027 * All filtered variables appear within the source text using a template
028 * and variable keys marked like <code>${example}</code>. When the XML 
029 * source file is read all template variables are replaced with the 
030 * values provided by the filter. If no replacement exists then the XML
031 * source text remains unchanged.
032 * 
033 * @author Niall Gallagher
034 */
035public interface Filter {
036
037   /**
038    * Replaces the text provided with some property. This method 
039    * acts much like a the get method of the <code>Map</code>
040    * object, in that it uses the provided text as a key to some 
041    * value. However it can also be used to evaluate expressions
042    * and output the result for inclusion in the generated XML.
043    *
044    * @param text this is the text value that is to be replaced
045    * 
046    * @return returns a replacement for the provided text value
047    */
048   String replace(String text);        
049}
