019package org.simpleframework.xml.filter;
020
021import java.util.Map;
022
023/**
024 * The <code>MapFilter</code> object is a filter that can make use
025 * of user specified mappings for replacement. This filter can be
026 * given a <code>Map</code> of name value pairs which will be used
027 * to resolve a value using the specified mappings. If there is
028 * no match found the filter will delegate to the provided filter. 
029 * 
030 * @author Niall Gallagher
031 */
032public class MapFilter implements Filter {
033   
034   /**
035    * This will resolve the replacement if no mapping is found.
036    */
037   private Filter filter;        
038
039   /**
040    * This contains a collection of user specified mappings.
041    */
042   private Map map;
043        
044   /**
045    * Constructor for the <code>MapFilter</code> object. This will
046    * use the specified mappings to resolve replacements. If this
047    * map does not contain a requested mapping null is resolved.
048    * 
049    * @param map this contains the user specified mappings
050    */
051   public MapFilter(Map map) {
052      this(map, null);           
053   }
054     
055   /**
056    * Constructor for the <code>MapFilter</code> object. This will
057    * use the specified mappings to resolve replacements. If this
058    * map does not contain a requested mapping the provided filter
059    * is used to resolve the replacement text.
060    * 
061    * @param map this contains the user specified mappings
062    * @param filter this is delegated to if the map fails
063    */   
064   public MapFilter(Map map, Filter filter) {
065      this.filter = filter;  
066      this.map = map;      
067   }
068
069   /**
070    * Replaces the text provided with the value resolved from the
071    * specified <code>Map</code>. If the map fails this will
072    * delegate to the specified <code>Filter</code> if it is not
073    * a null object. If no match is found a null is returned.
074    * 
075    * @param text this is the text value to be replaced
076    * 
077    * @return this will return the replacement text resolved
078    */
079   public String replace(String text) {
080      Object value = null;
081      
082      if(map != null) {
083         value = map.get(text);
084      }
085      if(value != null) {
086         return value.toString();                       
087      }
088      if(filter != null) {
089         return filter.replace(text);              
090      }      
091      return null;
092   }   
093}
