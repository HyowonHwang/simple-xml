019package org.simpleframework.xml.stream;
020
021/**
022 * The <code>Verbosity</code> enumeration is used to specify a verbosity
023 * preference for the resulting XML. Typically the verbosity preference
024 * is used when serializing an object that does not have explicit XML
025 * annotations associated with a type. In such a scenario this will
026 * indicate whether a high verbosity level is required or a low one.
027 * 
028 * @author Niall Gallagher
029 * 
030 * @see org.simpleframework.xml.stream.Format
031 */
032public enum Verbosity {
033   
034   /**
035    * This specifies a preference for elements over attributes.
036    */
037   HIGH,
038   
039   /**
040    * This specifies a preference for attributes over elements.
041    */
042   LOW;
043}
